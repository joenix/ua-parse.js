# ua-parse.js

```sh
yarn add ua-parse.js
# or
npm i ua-parse.js
```

```js
// Import Package
import uaParse from "ua-parse";

// Instantiate
const Parser = new uaParse();

// Get Full Infomation
Parser.get();

// Get System
Parser.getSystem();

// Get Engine
Parser.getEngine();

// Get Browser
Parser.getBrowser();

// Get Shell
Parser.getShell();

// Get Platform
Parser.getPlatform();
```
